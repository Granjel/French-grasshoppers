Plot total damage ~ grasshoppers:

I’m plotting now the total damage versus the amount of grasshoppers for each grasshopper species.

# Installing the ggplot2 package for cool graphics and gridExtra for grids

library(lattice)
Cb_plot <- qplot(x = Cb, y = Total, xlab = "Chorthippus biguttulus (Cb)", ylab = "Total herbivory", data = herbivory)
Cd_plot <- qplot(x = Cd, y = Total, xlab = "Chorthippus dorsatus (Cd)", ylab = " ", data = herbivory)
Ci_plot <- qplot(x = Ci, y = Total, xlab = "Calliptamus italicus (Ci)", ylab = " ", data = herbivory)
Ee_plot <- qplot(x = Ee, y = Total, xlab = "Euchorthippus elegantulus (Ee)", ylab = "Total herbivory", data = herbivory)
Pg_plot <- qplot(x = Pg, y = Total, xlab = "Pezotettix giornae (Pg)", ylab = " ", data = herbivory)
Pp_plot <- qplot(x = Pp, y = Total, xlab = "Pseudochorthippus parallelus (Pp)", ylab = " ", data = herbivory)
herbivory_grasshoppers <- grid.arrange(Cb_plot, Cd_plot, Ci_plot, Ee_plot, Pg_plot, Pp_plot, nrow = 2, ncol = 3)
